Creating DAX Measuring 12

Creating the necessary measures

                                

1.     Click on suicide data table and select New Measure

2.     Start writing the DAX formula (all the formulas are at the bottom) or copy and paste one by one.

3.     After you done click on the green right mark so it will be saved.

4.     Repeat the steps until all the measures are created.

.Arrow population Rate = SWITCH(TRUE(),[.Population YOY] < 0,UNICHAR(9662),[.Population YOY] = 0, UNICHAR(9666),UNICHAR(9652))

 

.Arrow Suicide Rate = SWITCH(TRUE(),[.YoY Total Suicide Rate (%)] < 0, UNICHAR(9662),[.YoY Total Suicide Rate (%)] = 0, UNICHAR(9662),UNICHAR(9652))

 

.Arrow Suicide YOY% = SWITCH(TRUE(),[Suicide YOY] < 0, UNICHAR(9662),[Suicide YOY] = 0, UNICHAR(9662),UNICHAR(9652))

 

.Average population = AVERAGEX('Suicide Data', [.TotalPopulation])

 

.Average Suicides = AVERAGE('Suicide Data'[Suicides_No])

 

.Female Suicide % = DIVIDE([Female],[.Total Suicides],0)

 

.Male Suicide % = DIVIDE([Male],[.Total Suicides],0)

 

.Population YOY =

VAR CurrentYear = MAX('CalendarTable'[Date].[Year])

VAR CurrentYearPopulation = [.Average population]

VAR PrvYear = CurrentYear - 1

VAR PrvYearPopulation =

    CALCULATE(

        [.Average population],

        FILTER(

            ALL('CalendarTable'[Date].[Year]),  -- Remove any date filters for accurate comparison

            'CalendarTable'[Date].[Year] = PrvYear

        )

    )

RETURN

IF(

    ISBLANK(PrvYearPopulation),

    BLANK(),  -- Return a blank if no data for the previous year

    DIVIDE(CurrentYearPopulation - PrvYearPopulation, PrvYearPopulation)

)

 

.Suicide Ratio by Age and Gender =

VAR MaleSuicides = CALCULATE(SUM('Suicide Data'[suicides_no]), 'Suicide Data'[sex] = "male")

VAR FemaleSuicides = CALCULATE(SUM('Suicide Data'[suicides_no]), 'Suicide Data'[sex] = "female")

 

RETURN

DIVIDE(

    MaleSuicides,

    FemaleSuicides,

    0

)

 

.Total Suicide Rate = [.Total Suicides] / [.TotalPopulation]

 

.Total Suicides = SUM('Suicide Data'[suicides_no])

 

.TotalPopulation =

VAR MaxYear = MAX('CalendarTable'[Date].[Year])

RETURN

    CALCULATE(SUM('Suicide Data'[population]), 'CalendarTable'[Date].[Year] = MaxYear)

 

.YoY Total Suicide Rate (%) =

VAR CurrentYear = MAX('CalendarTable'[Date].[Year])

VAR PrvYear = CurrentYear - 1

 

VAR TotalSuicideRateCurrentYear =

    CALCULATE(

        [.Total Suicide Rate],

        'CalendarTable'[Date].[Year] = CurrentYear

    )

 

VAR TotalSuicideRatePreviousYear =

    CALCULATE(

        [.Total Suicide Rate],

        'CalendarTable'[Date].[Year] = PrvYear

    )

 

RETURN

DIVIDE(

    TotalSuicideRateCurrentYear - TotalSuicideRatePreviousYear,

    TotalSuicideRatePreviousYear,

    0

)

 

Average Suicide Rate (%) =

AVERAGEX(

    SUMMARIZE('Suicide Data', 'Suicide Data'[Year]),

    DIVIDE(

        SUM('Suicide Data'[suicides_no]),

        SUM('Suicide Data'[population])

    )

)

 

Female = CALCULATE([.Total Suicides], 'Suicide Data'[sex] = "Female")

 

Male = CALCULATE([.Total Suicides], 'Suicide Data'[sex] = "Male")

 

Suicide YOY =

VAR CurrentYear = MAX('CalendarTable'[Date].[Year])

VAR CurrentYearSuicides = [.Average Suicides]

VAR PrvYear = CurrentYear - 1

VAR PrvYearSuicides =

    CALCULATE(

        [.Average Suicides],

        FILTER(

            ALL('CalendarTable'[Date].[Year]),  -- Remove any date filters for accurate comparison

            'CalendarTable'[Date].[Year] = PrvYear

        )

    )

RETURN

IF(

    ISBLANK(PrvYearSuicides),

    BLANK(),  -- Return a blank if no data for the previous year

    DIVIDE(CurrentYearSuicides - PrvYearSuicides, PrvYearSuicides)

)
